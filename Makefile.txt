# SpyK3-Bot Makefile

.PHONY: help install run docker-build docker-run clean test lint setup

# Colors
GREEN := \033[0;32m
YELLOW := \033[1;33m
NC := \033[0m

help:
	@echo "${GREEN}SpyK3-Bot Makefile Commands:${NC}"
	@echo "  make install     - Install dependencies"
	@echo "  make run         - Run the bot"
	@echo "  make setup       - Run setup script"
	@echo "  make docker-build- Build Docker image"
	@echo "  make docker-run   - Run in Docker container"
	@echo "  make test        - Run tests"
	@echo "  make lint        - Run linters"
	@echo "  make clean       - Clean temporary files"
	@echo "  make update      - Update dependencies"

install:
	@echo "${YELLOW}Installing dependencies...${NC}"
	python3 -m venv venv
	. venv/bin/activate && pip install --upgrade pip
	. venv/bin/activate && pip install -r requirements.txt
	@echo "${GREEN}✓ Installation complete${NC}"

run:
	@echo "${YELLOW}Starting SpyK3-Bot...${NC}"
	./run.sh

setup:
	@echo "${YELLOW}Running setup...${NC}"
	./setup.sh

docker-build:
	@echo "${YELLOW}Building Docker image...${NC}"
	docker build -t spyk3-bot:latest .
	@echo "${GREEN}✓ Docker image built${NC}"

docker-run:
	@echo "${YELLOW}Running in Docker...${NC}"
	docker run -it --rm \
		--cap-add=NET_ADMIN \
		--cap-add=NET_RAW \
		--network host \
		-v $(PWD)/.spyk3_bot:/app/.spyk3_bot \
		spyk3-bot:latest

test:
	@echo "${YELLOW}Running tests...${NC}"
	. venv/bin/activate && pytest tests/ -v

lint:
	@echo "${YELLOW}Running linters...${NC}"
	. venv/bin/activate && flake8 spyk3_bot.py --max-line-length=120 || true
	. venv/bin/activate && pylint spyk3_bot.py --exit-zero || true
	@echo "${GREEN}✓ Linting complete${NC}"

clean:
	@echo "${YELLOW}Cleaning up...${NC}"
	rm -rf __pycache__/
	rm -rf .pytest_cache/
	rm -rf .coverage
	rm -rf coverage/
	rm -rf .mypy_cache/
	rm -rf temp/
	find . -type d -name "__pycache__" -exec rm -rf {} + 2>/dev/null || true
	@echo "${GREEN}✓ Cleanup complete${NC}"

update:
	@echo "${YELLOW}Updating dependencies...${NC}"
	. venv/bin/activate && pip install --upgrade -r requirements.txt
	@echo "${GREEN}✓ Update complete${NC}"
